// add global git configuration
git config --global user.name "Andrew"

// show config
git config --list

// initialize local git repo
git init

// add to staging area
git add -A

// remove from staging area
git reset

// make commit
git commit -m "message here"

// show log
git log

// clone project from remote repo
git clone url .

// get info about remote
git remote -v

// show branch
git branch -a

// show changes between code
git diff
